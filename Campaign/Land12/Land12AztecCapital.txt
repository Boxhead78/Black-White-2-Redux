        mig12 = 1
        MigrationsAccepted = MigrationsAccepted + 1
    end if

    if MigrationsAccepted >= 1 and a == 0
        a = 1
        increment tribute by 5000
    end if

        a = 2
        increment tribute by 10000
    end if

    end if

        a = 7

    end if
    
end loop

begin script LC12_GetNearestSettlement

        NearestTown = AztecCapital

        NearestTown = AztecTown1

        NearestTown = AztecTown2

    elsif PLAYER of MinorTown1 == 0

    elsif PLAYER of MinorTown2 == 0
        NearestTown = MinorTown2

    elsif PLAYER of MinorTown3 == 0

    elsif PLAYER of MinorTown4 == 0
        NearestTown = MinorTown4
    else
        NearestTown = PlayerTown

    end if

    NearestSettlement = NearestTown
    wait 45 seconds
end loop

end script LC12_GetNearestSettlement

begin script DisableLocalPlatoonResponse

    Counter = 0

start

begin loop
    if Counter > 99
        Counter = 0
        wait 30 seconds
    end if
    if LC12_Platoon[Counter] exists
        disable platoon LC12_Platoon[Counter] respond to local platoon attack
    end if
    if LC12_WallArcher[Counter] exists
        disable platoon LC12_WallArcher[Counter] respond to local platoon attack
    end if

    Counter++
end loop

end script DisableLocalPlatoonResponse
