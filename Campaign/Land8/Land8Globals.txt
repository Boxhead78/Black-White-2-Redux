begin script LC8_DynamicDifficulty

    DifficultyTimer = create timer for 1500 seconds

start

begin loop
    wait 2.5 seconds
    if get DifficultyTimer time remaining <= 0
        RecruitModifier += 0.75
        EpicModifier += 0.25
        BuilderModifier += 0.33
        DifficultyTimer = create timer for 600 * Difficulty seconds
    end if
end loop

end script LC8_DynamicDifficulty
