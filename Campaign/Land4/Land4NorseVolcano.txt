begin script LC4_VolcanoFireBalls(Volcano)

    Number = 0
    FireBall = 0
    FireBallTarget = 0
    FireBallStart = 0

start

begin loop

    FireBallStart = marker at ({Volcano} + {0,30,0})
    FireBallTarget = marker at ({FireBallStart} + {number from -60 to 60, 0, number from -60 to 60})
    FireBall = make player 1 throw miracle MIRACLE_TYPE_FIRE from ({FireBallStart} + {0,-3.0,0}) heading ({FireBallTarget} + {0,21,0} - {FireBallStart})
    
    Number = number from 1 to 3
    
    if Number == 1
        wait 1.5 seconds
    elsif Number == 2
        wait 2.75 seconds
    elsif Number == 3
        wait 4.4 seconds
    end if
    
    until VolcanoStopFireBalls == 1
end loop

end script LC4_VolcanoFireBalls
