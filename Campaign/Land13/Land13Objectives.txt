define LO_FALSE = 0
define LO_TRUE = 1
define script LC13_TownObjectives

//-----------------------------------------------------------------------------
//    Set the objectives and test for when they are completed - passed back from AI script
//-----------------------------------------------------------------------------
begin script LC13_TownObjectives

    CreatureAbodeObjective = LO_FALSE
    CityObjective = LO_FALSE
    CreatureWoodObjective = LO_FALSE
    CreatureFoodObjective = LO_FALSE
    CreatureSoldiersObjective = LO_FALSE
    BreedObjective = LO_FALSE
    CreatureSoldiers2Objective = LO_FALSE
    PlayerAlignment = 0
    
start

    reset all objectives for player 0
    
    ///////PRIMARY OBJECTIVES - these are script controlled

    //PRIMARY - Dominate Land by Impressiveness or Force
    set player 0 objective TRIBUTE_OBJECTIVE_SCRIPT_CONTROLLED_WIN_LAND with amount 1 text "BW2T_SCRIPT_07FINAL_SECONDARYOBJECTIVE_80" amount 1 description "BW2T_SCRIPT_06FINAL_SECONDARYOBJECTIVE_51" start value 0 reward 450000 alignment 0.1
    
    /////SECONDARY OBJECTIVES - these are controlled by gamecode    
    //RESOURCE Objectives
    
    //Collect Resources    
    set player 0 objective TRIBUTE_OBJECTIVE_COLLECT_ALL with amount 36000 text "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_120" amount 36000 description "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_121" start value 0 reward 0
    
    //Collect ore    
    set player 0 objective TRIBUTE_OBJECTIVE_MINE_ORE with amount 12000 text "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_30" amount 12000 description "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_31" parent TRIBUTE_OBJECTIVE_COLLECT_ALL start value 0 reward 3000
    
    //Collect food    
    set player 0 objective TRIBUTE_OBJECTIVE_COLLECT_FOOD with amount 12000 text "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_90" amount 12000 description "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_91" parent TRIBUTE_OBJECTIVE_COLLECT_ALL start value 0 reward 3000
    
    //Collect wood    
    set player 0 objective TRIBUTE_OBJECTIVE_COLLECT_WOOD with amount 12000 text "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_100" amount 12000 description "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_101" parent TRIBUTE_OBJECTIVE_COLLECT_ALL start value 0 reward 3000
    
    
    //SPECIAL OBJECTIVES
    
    //Deforest the land
    set player 0 objective TRIBUTE_OBJECTIVE_AS_SOON_AS_TREES_REMOVED with amount 1 text "BW2T_SCRIPT_10FINAL_OBJECTIVE_EVIL_10" amount 1 description "BW2T_SCRIPT_10FINAL_OBJECTIVE_EVIL_11" start value 0 reward 30000 alignment -0.1
    
    //Increase impressiveness by 30000    
    set player 0 objective TRIBUTE_OBJECTIVE_INCREASE_IMPRESSIVENESS with amount 35000 text "BW2T_SCRIPT_06FINAL_OBJECTIVE_NEUTRAL_10" amount 35000 description "BW2T_SCRIPT_06FINAL_OBJECTIVE_NEUTRAL_11" start value 0 reward 15000 alignment 0.05
    
    
    //TOWN OBJECTIVES
    
    //Create 60 disciple breeders    
    set player 0 objective TRIBUTE_OBJECTIVE_MAKE_BREEDER_DISCIPLES with amount 60 text "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_50" amount 60 description "BW2T_SCRIPT_04FINAL_OBJECTIVE_NEUTRAL_51" start value 0 reward 20000


    //CREATURE OBJECTIVES

    //Creature Collects wood    
    set player 0 objective TRIBUTE_OBJECTIVE_CREATURE_COLLECT_WOOD with amount 10000 text "BW2T_SCRIPT_07FINAL_OBJECTIVE_NEUTRAL_80" amount 10000 description "BW2T_SCRIPT_07FINAL_OBJECTIVE_NEUTRAL_81" class TRIBUTE_OBJECTIVE_CLASS_CREATURE start value 0 reward 6000
    
    //Creature kills 10 platoons    
    set player 0 objective TRIBUTE_OBJECTIVE_CREATURE_KILL_PLATOONS with amount 10 text "BW2T_SCRIPT_07FINAL_OBJECTIVE_EVIL_50" amount 10 description "BW2T_SCRIPT_07FINAL_OBJECTIVE_EVIL_51" start value 0 reward 10000

    //Use creature to break enemy walls
    set player 0 objective TRIBUTE_OBJECTIVE_AS_SOON_AS_CREATURE_BREAK_ENEMY_WALLS with amount 1 text "BW2T_SCRIPT_OBJECTIVES_PICKUP29" amount 1 description "BW2T_SCRIPT_10FINAL_OBJECTIVE_PRIMARY_NEUTRAL_11" start value 0 reward 10000
    
    //Creature wins X creature fights 
    set player 0 objective TRIBUTE_OBJECTIVE_WIN_CREATURE_FIGHT with amount 3 text "BW2T_SCRIPT_OBJECTIVES_PICKUP32" amount 3 description "BW2T_SCRIPT_06FINAL_OBJECTIVE_EVIL_41" start value 0 reward 15000

    //Take over N towns by force 
    set player 0 objective TRIBUTE_OBJECTIVE_TAKEOVER_TOWNS_FORCE with amount 2 text "BW2T_SCRIPT_GENERIC_PICKUP162" amount 2 description "BW2T_SCRIPT_06FINAL_OBJECTIVE_EVIL_21" class TRIBUTE_OBJECTIVE_CLASS_ARMY start value 0 reward 20000   alignment -0.05
    
    //Take over X towns by impressiveness    
    set player 0 objective TRIBUTE_OBJECTIVE_TAKEOVER_TOWNS_IMPRESSIVENESS with amount 4 text "BW2T_SCRIPT_GENERIC_PICKUP163" amount 4 description "BW2T_SCRIPT_07FINAL_OBJECTIVE_GOOD_41" start value 0 reward 20000 alignment 0.05

end script LC13_TownObjectives
