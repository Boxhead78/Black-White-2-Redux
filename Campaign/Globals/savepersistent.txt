    RDX_PersistentDataSaved = 0

//Check if persistent data is lost, if it is lost try to get it from local variables
begin loop
    RDX_PersistentDataSaved = persistent data "RDX_PERSSAVED"
    if RDX_PersistentDataSaved != 1 and PD_ForcedSaving == 0
        PD_ForcedSaving = 1
        wait until PD_ForcedSaving == 0 or 10 seconds
        stop script "SavePersistentDataSetData"
        run script SavePersistentDataGetData
    end if
    wait 0.5 seconds
end loop
