///////////////////////////////////////////////////////////////////////////////
//          Skirmish Music System
//    ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
//
//    This file controls when which music plays during a skirmish match.
//
///////////////////////////////////////////////////////////////////////////////

define script SK_Music

begin script SK_Music

    Number = 0
    NumberOfPlatoons = 0
    Platoon = 0
    musicmixer = 0

    Counter = 0
    Town_ID[16]

    MusicIsRunning = 0

    PlayerAlignment = 0
    IntroMusic = 0

start

wait until LoadingScreenOver == 1

PlayerAlignment = get player 0 alignment
if PlayerAlignment < 0
    Number = number from 1 to 5
    if Number == 1
        start music "cut_scene_end_battle 1" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 114 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    elsif Number == 2
        start music "cut_scene_mystery_01" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 54 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    elsif Number == 3
        start music "cut_scene_sinister_01" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 74 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    elsif Number == 4
        start music "cut_scene_sinister_03" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 57 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    else
        start music "nemesis" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 90 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    end if

elsif PlayerAlignment >= 0
    Number = number from 1 to 5
    if Number == 1
        start music "cut_scene_celtic_air" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 120 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    elsif Number == 2
        start music "cut_scene_mystery_01" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 54 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    elsif Number == 3
        start music "cut_scene_sad_100_01" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 50 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    elsif Number == 4
        start music "cut_scene_the_awakening" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 120 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    else
        start music "khazar" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1
        wait 54 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    end if

end if
IntroMusic = 1

wait 300 seconds

run background script DMM_DynamicMusicManager(DMM_MUSIC_PAUSE_ON, DMM_SERVERITY_LEVEL_4)

end script SK_Music
