    NumberOfPlatoons = 0

    musicmixer = 0

    wait 90 seconds
end if

    if Number == 1
        start music "cut_scene_end_battle 1" loop -1
        wait 2 seconds
        SK_MusicIsRunning = 1

        SK_MusicIsRunning = 0
        wait 2.5 seconds
        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

        start music "cut_scene_mystery_01" loop -1

        wait 15 seconds

        SK_MusicIsRunning = 1
        wait 90 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds

        wait 15 seconds

elsif PlayerAlignment >= 0

    if Number == 1
        start music "cut_scene_celtic_air" loop -1

        SK_MusicIsRunning = 0

        if SK_AltarMusicCounter == 0 stop music with fadetime 9 end if
        wait 15 seconds

    elsif Number == 2
        start music "cut_scene_mystery_01" loop -1
        wait 2 seconds

        SK_MusicIsRunning = 1
        wait 54 seconds
        SK_MusicIsRunning = 0
        wait 2.5 seconds

        wait 15 seconds

    if MusicIsRunning == 0

            NumberOfPlatoons = 0
            Platoon = get platoon of player 1 nearest {Town_ID[Counter]} radius 400
            NumberOfPlatoons = get number of platoon of player 1 at {Town_ID[Counter]} radius 400

            if Platoon exists and get number of villagers in platoon Platoon > 100 and MusicIsRunning == 0
                MusicIsRunning = 1
                Number = number from 0 to 3
                start dynamic music
                set dynamic music "small" loop -1 urgent
                wait 2 seconds
                SK_MusicIsRunning = 1
                SK_DynamicMusicIsRunning = 1

                if Number == 0
                    set dynamic music "largeevent1" loop -1 urgent
                    wait 30 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 1
                    set dynamic music "smallevent4" loop 0 urgent
                    wait 8 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 2
                    set dynamic music "medium" loop -1

                elsif Number == 3
                    set dynamic music "large" loop -1
                    wait 45 seconds
                    set dynamic music "medium" loop -1

                end if
            elsif NumberOfPlatoons >= 4 and MusicIsRunning == 0
                MusicIsRunning = 1
                Number = number from 0 to 3

                set dynamic music "small" loop -1 urgent

                SK_DynamicMusicIsRunning = 1

            NumberOfPlatoons = 0
            Platoon = get platoon of player 2 nearest {Town_ID[Counter]} radius 400
            NumberOfPlatoons = get number of platoon of player 2 at {Town_ID[Counter]} radius 400

            if Platoon exists and get number of villagers in platoon Platoon > 100 and MusicIsRunning == 0
                MusicIsRunning = 1
                Number = number from 0 to 3
                start dynamic music
                set dynamic music "small" loop -1 urgent
                wait 2 seconds
                SK_MusicIsRunning = 1
                SK_DynamicMusicIsRunning = 1
                wait 40 seconds
                if Number == 0
                    set dynamic music "largeevent1" loop -1 urgent
                    wait 30 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 1
                    set dynamic music "smallevent4" loop 0 urgent
                    wait 8 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 2
                    set dynamic music "medium" loop -1

                elsif Number == 3
                    set dynamic music "large" loop -1
                    wait 45 seconds
                    set dynamic music "medium" loop -1

                end if
            elsif NumberOfPlatoons >= 4 and MusicIsRunning == 0
                MusicIsRunning = 1
                Number = number from 0 to 3
                start dynamic music
                set dynamic music "small" loop -1 urgent
                wait 2 seconds
                SK_MusicIsRunning = 1
                SK_DynamicMusicIsRunning = 1
                wait 40 seconds
                if Number == 0
                    set dynamic music "largeevent1" loop -1 urgent
                    wait 30 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 1
                    set dynamic music "smallevent4" loop 0 urgent
                    wait 8 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 2
                    set dynamic music "medium" loop -1

                elsif Number == 3
                    set dynamic music "large" loop -1
                    wait 45 seconds
                    set dynamic music "medium" loop -1

                end if
            end if

            Platoon = 0
            NumberOfPlatoons = 0
            Platoon = get platoon of player 3 nearest {Town_ID[Counter]} radius 400
            NumberOfPlatoons = get number of platoon of player 3 at {Town_ID[Counter]} radius 400

            if Platoon exists and get number of villagers in platoon Platoon > 100 and MusicIsRunning == 0
                MusicIsRunning = 1
                Number = number from 0 to 3
                start dynamic music
                set dynamic music "small" loop -1 urgent
                wait 2 seconds
                SK_MusicIsRunning = 1
                SK_DynamicMusicIsRunning = 1
                wait 40 seconds
                if Number == 0
                    set dynamic music "largeevent1" loop -1 urgent
                    wait 30 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 1
                    set dynamic music "smallevent4" loop 0 urgent
                    wait 8 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 2
                    set dynamic music "medium" loop -1

                elsif Number == 3
                    set dynamic music "large" loop -1
                    wait 45 seconds
                    set dynamic music "medium" loop -1

                end if
            elsif NumberOfPlatoons >= 4 and MusicIsRunning == 0
                MusicIsRunning = 1
                Number = number from 0 to 3
                start dynamic music
                set dynamic music "small" loop -1 urgent
                wait 2 seconds
                SK_MusicIsRunning = 1
                SK_DynamicMusicIsRunning = 1

            Platoon = get platoon of player 0 nearest {Town_ID[Counter]} radius 400
            NumberOfPlatoons = get number of platoon of player 0 at {Town_ID[Counter]} radius 400

                set dynamic music "small" loop -1 urgent
                wait 2 seconds
                SK_MusicIsRunning = 1
                SK_DynamicMusicIsRunning = 1
                wait 40 seconds
                if Number == 0
                    set dynamic music "largeevent1" loop -1 urgent
                    wait 30 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 1
                    set dynamic music "smallevent4" loop 0 urgent
                    wait 8 seconds
                    set dynamic music "medium" loop -1

                elsif Number == 2
                    set dynamic music "medium" loop -1

                elsif Number == 3
                    set dynamic music "large" loop -1
                    wait 45 seconds
                    set dynamic music "medium" loop -1

                end if
            elsif NumberOfPlatoons > 3 and MusicIsRunning == 0

                set dynamic music "small" loop -1 urgent
                wait 2 seconds

                wait 40 seconds
